### 70. Climbing Stairs

```
class Solution:
    def climbStairs(self, n: int) -> int:
        dp = [0]*46
        dp[:3] = [0, 1, 2]
        for i in range(3, len(dp)):
            dp[i] = dp[i-1] + dp[i-2]
        return dp[n]
```

### 118. Pascal's Triangle

```
class Solution:
    def generate(self, numRows: int) -> List[List[int]]:
        if numRows == 1:
            return [[1]]
        else:
            pascal = [[1]]
            for i in range(1,numRows):
                temp = [1]
                left, right = 0, 1
                while right < i:
                    a = pascal[-1][left] + pascal[-1][right]
                    temp.append(a)
                    left, right = left +1, right + 1
                temp.append(1)
                pascal.append(temp)
            return pascal
```
