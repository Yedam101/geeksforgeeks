### 125. Valid Palindrome

```
class Solution:
    def isPalindrome(self, s: str) -> bool:
        news = ''.join(char.lower() for char in s if char.isalnum()) # **rmb
        left, right = 0, len(news) -1


        while left < right:
            if news[left] == news[right]:
                left, right = left + 1, right - 1
            else:
                return False
        return True

```

### 167. Two Sum II - Input Array Is Sorted

```
class Solution:
    def twoSum(self, numbers: List[int], target: int) -> List[int]:
        dict = {}
        for i, v in enumerate(numbers):
            if target - v in dict:
                return [min(i+1, dict[target - v]), max(i+1, dict[target - v])]
            dict[v] = i + 1
```

### 15. 3Sum

```
class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        result = []
        nums = sorted(nums)

        for i in range(len(nums)):
            l, r = i+1, len(nums)-1
            while l < r:
                if -nums[i] < nums[l] + nums[r]:
                    r -= 1
                elif -nums[i] > nums[l] + nums[r]:
                    l += 1
                elif -nums[i] == nums[l] + nums[r] and i!=l!=r:
                    p = [nums[i], nums[l], nums[r]]
                    if p not in result:
                        result.append(p)
                    l += 1
        return result
```

### 3. Longest Substring Without Repeating Characters

```
class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        if s == "":
            return 0
        else:
            temp = 1
            l, r = 0, 1
            while r < len(s):
                if s[r] not in s[l:r]:
                    r += 1
                else:
                    l,r = l+1, l+2
                temp = max(temp, len(s[l:r]))
            return temp
```
